SELECT CARGO, TRATAMENTO, COUNT(*)
FROM EMPREGADOS 
GROUP BY CARGO, TRATAMENTO
ORDER BY CARGO

SELECT CARGO, COUNT(*)
FROM EMPREGADOS
GROUP BY CARGO
HAVING COUNT(*) > 1
ORDER BY CARGO


SELECT NUMERODOPEDIDO, 
       SUM(PRECOUNITARIO*QUANTIDADE*(1-DESCONTO))
FROM DETALHES_PEDIDO 
GROUP BY NUMERODOPEDIDO
ORDER BY NUMERODOPEDIDO

SELECT NOMEDOPRODUTO, NOMEDACATEGORIA, PRECOUNITARIO
FROM PRODUTOS P INNER JOIN CATEGORIAS C
ON P.CODIGODACATEGORIA = C.CODIGODACATEGORIA
WHERE PRECOUNITARIO >= 50


SELECT NOMEDOPRODUTO, NOMEDACATEGORIA, PRECOUNITARIO
FROM PRODUTOS P ,CATEGORIAS C
WHERE P.CODIGODACATEGORIA = C.CODIGODACATEGORIA
AND PRECOUNITARIO >= 50

SELECT NOMEDOPRODUTO, NOMEDAEMPRESA, NOMEDACATEGORIA, PRECOUNITARIO
FROM PRODUTOS P INNER JOIN CATEGORIAS C
ON P.CODIGODACATEGORIA = C.CODIGODACATEGORIA
LEFT JOIN FORNECEDORES F
ON F.CODIGODOFORNECEDOR = P.CODIGODOFORNECEDOR
--WHERE PRECOUNITARIO >= 50


SELECT NOMEDOPRODUTO, NOMEDAEMPRESA, NOMEDACATEGORIA, PRECOUNITARIO
FROM PRODUTOS P ,CATEGORIAS C, FORNECEDORES F
WHERE P.CODIGODACATEGORIA = C.CODIGODACATEGORIA
AND F.CODIGODOFORNECEDOR = P.CODIGODOFORNECEDOR
AND PRECOUNITARIO >= 50
