SELECT * 
FROM EMPREGADOS
WHERE REGIAO is not null

--COMENTARIOS NINJA
SELECT NOMEDOPRODUTO, LOWER(NOMEDOPRODUTO)
FROM PRODUTOS
WHERE LOWER(NOMEDOPRODUTO) NOT LIKE 'c%'

SELECT *
FROM PRODUTOS
WHERE CODIGODOPRODUTO NOT IN (4,5,7,8)

SELECT DISTINCT CARGO
FROM EMPREGADOS

SELECT codigodofuncionario, cargo
FROM EMPREGADOS
where MOD(CODIGODOFUNCIONARIO,2) <> 0

SELECT COUNT(*)
FROM DETALHES_PEDIDO

SELECT COUNT(*), COUNT(REGIAO),
       COUNT(*) - COUNT(REGIAO)
FROM EMPREGADOS 

SELECT COUNT(*),MAX(PRECOUNITARIO), MIN(PRECOUNITARIO),
       ROUND(AVG(PRECOUNITARIO),2),
       SUM(UNIDADESEMESTOQUE*PRECOUNITARIO)
FROM PRODUTOS


SELECT CARGO, TRATAMENTO, COUNT(*)
FROM EMPREGADOS 
GROUP BY CARGO, TRATAMENTO

SELECT QUANTIDADEPORUNIDADE, COUNT(*)
FROM PRODUTOS 
GROUP BY QUANTIDADEPORUNIDADE
ORDER BY COUNT(*) DESC
