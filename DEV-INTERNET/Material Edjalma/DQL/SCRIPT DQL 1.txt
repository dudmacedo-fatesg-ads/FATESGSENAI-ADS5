SELECT CODIGODACATEGORIA,NOMEDACATEGORIA 
FROM CATEGORIAS
WHERE CODIGODACATEGORIA >= 5

SELECT NOMEDOPRODUTO AS PRODUTO, 
       TO_CHAR(PRECOUNITARIO,'L999.99') AS PREÇO, 
       TO_CHAR(PRECOUNITARIO*1.2,'L999.99') AS PREÇO2
FROM PRODUTOS
WHERE PRECOUNITARIO > 60
ORDER BY NOMEDOPRODUTO DESC

SELECT NOMEDOPRODUTO, PRECOUNITARIO
FROM PRODUTOS
WHERE PRECOUNITARIO >= 30 AND PRECOUNITARIO <= 35
AND NIVELDEREPOSICAO > 3

WHERE PRECOUNITARIO BETWEEN 30 AND 35

SELECT CODIGODOPRODUTO, NOMEDOPRODUTO
FROM PRODUTOS
WHERE CODIGODOPRODUTO = 4 OR
CODIGODOPRODUTO = 6 OR
CODIGODOPRODUTO = 7 OR
CODIGODOPRODUTO = 9;

SELECT CODIGODOPRODUTO, NOMEDOPRODUTO
FROM PRODUTOS
WHERE CODIGODOPRODUTO IN (4,6,7,9)
AND PRECOUNITARIO > 30;

SELECT NOMEDOPRODUTO
FROM PRODUTOS
WHERE UPPER(NOMEDOPRODUTO) = 'CHAI'

SELECT NOMEDOPRODUTO
FROM PRODUTOS
WHERE UPPER(NOMEDOPRODUTO) LIKE '%T%'

SELECT NOMEDOPRODUTO
FROM PRODUTOS
WHERE UPPER(NOMEDOPRODUTO) LIKE '__E%'

SELECT NOMEDOPRODUTO
FROM PRODUTOS
WHERE UPPER(NOMEDOPRODUTO) LIKE 'CHE%'
