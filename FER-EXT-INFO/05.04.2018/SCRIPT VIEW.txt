SELECT COUNT(*), MAX(PRECOUNITARIO), MIN(PRECOUNITARIO),
       ROUND(AVG(PRECOUNITARIO),2),
       TRUNC(AVG(PRECOUNITARIO),2),
       SUM(UNIDADESEMESTOQUE*PRECOUNITARIO)
FROM PRODUTOS;

CREATE VIEW RESUMO_PROD AS
SELECT COUNT(*) QUANT, MAX(PRECOUNITARIO)MAIOR_PREÇO, MIN(PRECOUNITARIO) MENOR_PREÇO,
       ROUND(AVG(PRECOUNITARIO),2)MEDIA_ROUND,
       TRUNC(AVG(PRECOUNITARIO),2) MEDIA_TRUNC,
       SUM(UNIDADESEMESTOQUE*PRECOUNITARIO) SOMA
FROM PRODUTOS;

SELECT *
FROM RESUMO_PROD;

CREATE VIEW CARGOS_EMP AS
SELECT CARGO,TRATAMENTO, COUNT(*) QUANT
FROM EMPREGADOS 
GROUP BY CARGO, TRATAMENTO;

SELECT * FROM CARGOS_EMP;
